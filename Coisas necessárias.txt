const imgs = document.querySelector('div#img')
const img = document.querySelectorAll("#img img")

let idx = 0

function carrossel(){
    idx++
    if(idx > img.length - 1) idx = 0
    
    $('.container > img').eq(idx).show()

    imgs.style.transform = `translateX(${-idx * $('.container').width()})`

    $('.container > img').eq(idx - 1).hide();

}
